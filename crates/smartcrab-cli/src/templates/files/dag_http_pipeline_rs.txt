use smartcrab::prelude::*;

use crate::layer::input::http_input::HttpInput;
use crate::layer::output::discord_output::DiscordOutput;

/// Build the HTTP pipeline graph:
///   HttpInput -> DiscordOutput
pub fn build() -> std::result::Result<DirectedGraph, GraphError> {
    DirectedGraphBuilder::new("http_pipeline")
        .add_input(HttpInput {
            target_channel_id: std::env::var("HTTP_CHANNEL_ID")
                .unwrap_or_else(|_| "000000000000000000".to_owned()),
        })
        .add_output(DiscordOutput)
        .add_edge("HttpInput", "DiscordOutput")
        .build()
}
