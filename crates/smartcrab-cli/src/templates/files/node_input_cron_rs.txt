use smartcrab::prelude::*;

use crate::dto::discord::DiscordNotification;

/// Input node that fires on a cron schedule (every minute).
pub struct CronInput {
    pub target_channel_id: String,
}

impl Node for CronInput {
    fn name(&self) -> &str {
        "CronInput"
    }
}

#[async_trait]
impl InputNode for CronInput {
    type TriggerData = ();
    type Output = DiscordNotification;

    async fn run(&self, _: ()) -> Result<DiscordNotification> {
        // Wait for the next tick (every 60 seconds)
        tokio::time::sleep(std::time::Duration::from_secs(60)).await;

        Ok(DiscordNotification {
            channel_id: self.target_channel_id.clone(),
            content: format!(
                "Scheduled ping at {}",
                std::time::SystemTime::now()
                    .duration_since(std::time::UNIX_EPOCH)
                    .unwrap_or_default()
                    .as_secs()
            ),
        })
    }
}
