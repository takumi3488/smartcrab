mod dag;
mod dto;
mod layer;

use smartcrab::prelude::*;

#[tokio::main]
async fn main() -> Result<()> {
    smartcrab::telemetry::init()?;

    tracing::info!("starting {{name}}");

    let discord_graph = dag::discord_pipeline::build()?;
    let cron_graph = dag::cron_pipeline::build()?;
    let http_graph = dag::http_pipeline::build()?;

    Runtime::new()
        .add_graph(discord_graph)
        .add_graph(cron_graph)
        .add_graph(http_graph)
        .run()
        .await
}
