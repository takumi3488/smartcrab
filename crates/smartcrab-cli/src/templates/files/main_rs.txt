mod dto;
mod graph;
mod layer;

use smartcrab::prelude::*;

#[tokio::main]
async fn main() -> Result<()> {
    smartcrab::telemetry::init()?;

    tracing::info!("starting {{name}}");

    let discord_graph = graph::discord_pipeline::build()?;
    let cron_graph = graph::cron_pipeline::build()?;

    Runtime::new()
        .add_graph(discord_graph)
        .add_graph(cron_graph)
        .run()
        .await
}
