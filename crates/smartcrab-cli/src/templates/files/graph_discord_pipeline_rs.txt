use smartcrab::prelude::*;

use crate::node::hidden::claude_code_node::ClaudeCodeNode;
use crate::node::input::discord_input::DiscordInput;
use crate::node::output::discord_output::DiscordOutput;

/// Build the Discord pipeline graph:
///   DiscordInput -> ClaudeCodeNode -> DiscordOutput
pub fn build() -> std::result::Result<DirectedGraph, GraphError> {
    DirectedGraphBuilder::new("discord_pipeline")
        .trigger(TriggerKind::discord(
            vec!["mention".into(), "dm".into()],
            None,
        ))
        .add_input(DiscordInput)
        .add_hidden(ClaudeCodeNode::new())
        .add_output(DiscordOutput)
        .add_edge("DiscordInput", "ClaudeCodeNode")
        .add_edge("ClaudeCodeNode", "DiscordOutput")
        .build()
}
