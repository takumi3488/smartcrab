use smartcrab::chat::discord::DiscordMessage;
use smartcrab::prelude::*;

/// Input node that receives Discord messages (mentions and DMs).
pub struct DiscordInput;

impl Node for DiscordInput {
    fn name(&self) -> &str {
        "DiscordInput"
    }
}

#[async_trait]
impl InputNode for DiscordInput {
    type TriggerData = DiscordMessage;
    type Output = DiscordMessage;

    async fn run(&self, msg: DiscordMessage) -> Result<DiscordMessage> {
        Ok(msg)
    }
}
